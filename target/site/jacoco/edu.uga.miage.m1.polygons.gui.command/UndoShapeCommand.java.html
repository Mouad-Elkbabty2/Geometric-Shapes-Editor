package edu.uga.miage.m1.polygons.gui.command;

import java.util.List;

import edu.uga.miage.m1.polygons.gui.JDrawingFrame;
import edu.uga.miage.m1.polygons.gui.shapes.SimpleShape;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
public class UndoShapeCommand implements Command {
    private JDrawingFrame jDrawingFrame;
    private List<Command> commandHistory;
    private List<SimpleShape> listOfShapes;
    private List<SimpleShape> lastSelectedShape;

    public UndoShapeCommand(JDrawingFrame jDrawingFrame, List<Command> commandHistory, List<SimpleShape> listOfShapes,
            List<SimpleShape> lastSelectedShape) {
        this.jDrawingFrame = jDrawingFrame;
        this.commandHistory = commandHistory;
        this.listOfShapes = listOfShapes;
        this.lastSelectedShape = lastSelectedShape;
    }

    public void execute() {
        if (!commandHistory.isEmpty() && !listOfShapes.isEmpty() && this.lastSelectedShape.size() == 0) {
            commandHistory.remove(commandHistory.size() - 1);
            listOfShapes.remove(listOfShapes.size() - 1);
        }
    }

}
